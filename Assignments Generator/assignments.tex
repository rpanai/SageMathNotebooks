$$
A=\left(\begin{array}{rrr}

\end{array}\right)
$$

\begin{solution}
$$A^{-1}=\left(\begin{array}{rrr}
\frac{1}{5} & -\frac{7}{5} & \frac{4}{5} \\
\frac{2}{5} & \frac{1}{5} & -\frac{2}{5} \\
-\frac{2}{5} & \frac{9}{5} & -\frac{3}{5}
\end{array}\right)$$
\end{solution}

$$
A=\left(\begin{array}{rrr}
-6 & -2 & 2 \\
-3 & -5 & 7 \\
-6 & -4 & 6
\end{array}\right)
$$
the matrix of $T$ with respect to the canonical basis.
\begin{parts}
\part Calculate $\det(A-\lambda I)$.
\part Find the eigenvalues for $A$.
\part Find the eigenvectors for $A$.
\part Is $A$ diagonalizable?
\end{parts}

\item ${\lambda}_1=2, {\lambda}_2=-3, {\lambda}_3=-4$.
\item $v_{2}=\left(\begin{array}{r}

\end{array}\right), v_{-3}=\left(\begin{array}{r}
1 \\
-\frac{3}{2} \\
0
\end{array}\right), v_{-4}=\left(\begin{array}{r}
1 \\
-\frac{2}{3} \\
\frac{1}{3}
\end{array}\right)$.
\item Yes.
\end{enumerate}
\end{solution}

\end{questions}
